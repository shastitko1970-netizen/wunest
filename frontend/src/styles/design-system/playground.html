Что менять, чтобы не сломать

/* ✅ Безопасно: меняет только значения */
:root {
  --SmartThemeBlurTintColor: #1a130a;
  --SmartThemeChatTintColor: #2a1f12;
  --SmartThemeBorderColor:   #6b4a2a;
  --SmartThemeBodyColor:     #f4e4c1;
  --SmartThemeQuoteColor:    #d4904a;
}

/* ✅ Безопасно: публичные якоря */
.nest-msg { border-radius: 4px }
.nest-msg-name { font-style: italic }
.mes { ... }           /* SillyTavern-алиас */
#chat { ... }          /* id-якорь */

/* ⚠ Осторожно: глобально ломает шелл */
body { font-family: Comic Sans !important }

/* ❌ Не делай так: приватный селектор из компонента */
.v-btn__overlay { opacity: 0 !important }
